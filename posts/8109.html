创建数据库

登入SSMS,连接Sql Server 后，点击新建查询，在弹出的面板中输入：

create database DBtest

或者创建数据库,自定义属性：

if exists (select * from sys.databases where name = 'DBtest')
	drop database DBtest  --简单粗暴，在工作中慎用。
	
create database DBtest
on  --数据文件 
(
	name = 'DBtest',--逻辑名称
	filename = 'C:\test\test3\DBtest.mdf', --物理路径
	size = 8MB , --初始大小
	filegrowth = 20%  --文件增长方式 或者 filegrowth = 2MB，即可以百分比设置，或者具体大小。

log on --日志文件
(
	name = 'DBtes_log',--逻辑名称
	filename = 'C:\test\test3\DBtest_log.mdf', --物理路径
	size = 8MB , --初始大小
	filegrowth = 20% --文件增长方式 或者 filegrowth = 2MB，即可以百分比设置，或者具体增长大小。

创建表

if exists(select*from sys.objects where name = 'Department' and type = 'U')
 drop table Department --超级危险，慎用

use DBtest --切换数据库

--创建表基本语法
create table Department(
 D_Id int primary key identity(1,1), --identity(1,1):自动增长，初始值1，每次增长1。
 D_Name nvarchar(50) not null,
 D_Remark text --部门描述
)    
--char:定长
--varchar:变长
--text:长文本(效率较低)
--char,varchar,text前面加n:存储unicode字符，对中文比较友好
--比如：varchar(100):存储100个字母或者50个汉字
--nvarchar(100):存储100个字母或者100个汉字

create table [Rank](
 R_Id int primary key identity(1,1), --identity(1,1):自动增长，初始值1，每次增长1。
 R_Name nvarchar(50) not null,
 R_Remark text     
)  

create table People (
 P_Id int primary key identity(1,1), --identity(1,1):自动增长，初始值1，每次增长1。
 D_Id int references Department(D_Id) not null , --引用外键
 R_Id int references [Rank](R_Id) not null ,   
 P_Name nvarchar(50) not null,
 P_Gender nvarchar(1) default('男') check(P_Gender = '男'or P_Gender = '女') not null ,
 P_Birthday smalldatetime not null,
 P_Salary decimal(12,2) check(P_Salary >= 1000 and  P_Salary <= 1000000) not null ,
 P_Phone varchar(20) unique not null,
 P_Address varchar(300),
 P_AddTime smalldatetime default(getdate())
)

维护约束(删除，添加)

--删除约束
--语法：alter table tableName drop constraint constrantName
--例子：删除月薪的约束
alter table People drop constraint CK__People__P_Salary__38996AB5

--添加约束(check约束)
alter table tableName add constraint constraintName check (表达式) 
--例子：添加工资字段约束，工资必须在1000-1000000之间
alter table People add constraint CK__People__P_Salary check (P_Salary >= 1000 and P_Salary <= 1000000)

--添加约束(主键约束) 语法：
alter table tableName add constraint constraintName  primary key(列名)

--添加约束(唯一约束) 语法：
alter table tableName add constraint constraintName  unique(列名)

--添加约束(默认值) 语法：
alter table tableName add constraint constraintName  default 默认值 for 列名 

--添加约束(外键) 语法：
alter table tableName add constraint constraintName  foreign key(列名) references 关联表名(列名)

drop,truncate,delete简单区别：

--关于删除(drop,truncate,delete)
delete from People --删除表数据，表对象和表结构依然存在
truncate table People --删除表数据(清空数据)，表对象和表结构依然存在
drop table People --删除表对象(删除最彻底)
--truncate ,delete 区别：
--truncate 清空所有数据，不能有条件。同时已经自动编号1，2，3，4 ...仍存在。即，原先表中有1，2，3，4.然后truncate 后，再添加数据，自动编号仍是1234
--delete 可以删除所有数据，也可以带条件删除符合条件的数据。同时，已经自动编号的编号将永远消失。即，原先表中有1，2，3，4.然后delete 后，再添加数据，自动编号将变成5，6，7，8

条件查询

use DBtest

select * from People where P_Gender = '女'

select * from People where P_Salary >= 10000

--查询出性别为女，工资大于等于10000元的员工信息（多样性）
select * from People where P_Gender = '女' and P_Salary >= 10000

--查询出出生年月在1980-1-1之后，而且月薪大于等于10000的女员工
select * from People where P_Birthday >= '1980-1-1' and P_Salary >= 10000 and P_Gender = '女'

--查询月薪大于等于10000的员工，或者月薪大于等于8000的女员工
select * from People where P_Salary >= 10000 or (P_Salary >= 8000 and P_Gender = '女')

select * from People where P_Salary between 10000 and 20000

select * from People where P_Address in ('北京','武汉')

select * from People order by len(P_Name) desc

模糊查询

%：代表匹配0个字符、1个字符或多个字符
_：代表匹配有且只有1个字符
[]：代表匹配范围内
[^]：代表匹配不在范围内

例子如下：

--选取数据库
use DBtest

--查询姓刘的员工信息
select * from People  where P_Name like '刘%'

--查询名字中含有'备'的员工信息
select * from People  where P_Name like '%备%'

--查询名字中含有'备'或者'刘'的员工信息
select * from People  where P_Name like '%备%' or P_Name like '%刘%'

--查询姓刘的员工，名字是2个字(方法一)
select * from People  where P_Name like '刘_'

--查询姓刘的员工，名字是2个字(方法二)
select * from People where substring(P_Name,1,1) = '刘' and len(P_Name) = 2

--查询出名字最后一个字是备，名字一共两个字的员工信息(方法一)
select * from People  where P_Name like '_备'

--查询出名字最后一个字是备，名字一共两个字的员工信息(方法二)
select * from People where substring(P_Name,2,1) = '备' and LEN(P_Name) = 2

--查询出电话号码开头138的员工信息
select * from People where P_Phone like '138%'

--查询出电话号码开头138的员工信息,第4位可能是5，可能8 ，最后一个号码是8
select * from People where P_Phone like '138[5,8]%8'

--查询出电话号码开头138的员工信息,第4位是2-5之间的数字 ，最后一个号码不是2和3(方法一)
select * from People where P_Phone like '138[2,3,4,5]%[^2,3]'

--查询出电话号码开头138的员工信息,第4位是2-5之间的数字 ，最后一个号码不是2和3(方法二)
select * from People where P_Phone like '138[2-5]%[^2-3]'

聚合函数

count:求数量
max:求最大值
min:求最小值
sum:求和
avg:求平均值

--选取数据库
use DBtest

--求员工总人数
select COUNT(*) Count_Num from People

--求最大值，求最高工资
select MAX(P_Salary) Max_P_Salary from People

--求最小时，求最小工资
select MIN(P_Salary) Min_P_Salary from People

--求和，求所有员工的工资总和
select SUM(P_Salary) Sum_P_Salary from People

--求平均值，求所有员工的平均工资(方法一)
select AVG(P_Salary) Avg_P_Salary from People

--求平均值，求所有员工的平均工资(方法二)
select ROUND(AVG(P_Salary),2) Avg_P_Salary from People

----求平均值，求所有员工的平均工资(方法三)
select Convert(decimal(12,2),AVG(P_Salary))  Avg_P_Salary from People

--求数量，最大值，最小值，总和，平均值，在一行显示
select COUNT(*) Count_Num ,MAX(P_Salary) Max_P_Salary,MIN(P_Salary) Min_P_Salary ,SUM(P_Salary) Sum_P_Salary,AVG(P_Salary) Avg_P_Salary from People

--查询出中国的员工人数，总工资，最高工资，最低工资和平均工资
select '中国' country, COUNT(*) Count_Num ,SUM(P_Salary) Sum_P_Salary ,MAX(P_Salary) Max_P_Salary, MIN(P_Salary) Min_P_Salary , AVG(P_Salary) Avg_P_Salary from People where P_Address='中国'

--求出工资比平均工资高的人员信息
select * from People where P_Salary > (select avg(P_Salary) from People)

--求数量，年龄最大值，年龄最小值，年龄总和，年龄平均值，在一行显示(方法一)
select COUNT(*) Count_Num,
 MAX(year(getdate())-year(P_Birthday)) Max_P_Birthady,
 MIN(year(getdate())-year(P_Birthday)) Min_P_Birthady,
 SUM(year(getdate())-year(P_Birthday)) Sum_P_Birthady,
 AVG(year(getdate())-year(P_Birthday)) Avg_P_Birthady 
from People

--求数量，年龄最大值，年龄最小值，年龄总和，年龄平均值，在一行显示(方法二)
select COUNT(*) Count_Num,
 MAX(DATEDIFF(year, P_Birthday, getDate())) Max_P_Birthady,
 MIN(DATEDIFF(year, P_Birthday, getDate())) Min_P_Birthady,
 SUM(DATEDIFF(year, P_Birthday, getDate())) Sum_P_Birthady,
 AVG(DATEDIFF(year, P_Birthday, getDate())) Avg_P_Birthady 
from People

--计算出月薪在10000 以上的男性员工的最大年龄，最小年龄和平均年龄(方法一)
select '男' Gender,COUNT(*) Count_Num,
 MAX(year(getdate())-year(P_Birthday)) Max_P_Birthady,
 MIN(year(getdate())-year(P_Birthday)) Min_P_Birthady,

select '男' Gender,COUNT(*) Count_Num,
 MAX(DATEDIFF(year, P_Birthday, getDate())) Max_P_Birthady,
 MIN(DATEDIFF(year, P_Birthday, getDate())) Min_P_Birthady,
 SUM(DATEDIFF(year, P_Birthday, getDate())) Sum_P_Birthady,
 AVG(DATEDIFF(year, P_Birthday, getDate())) Avg_P_Birthady 
from People where P_Gender = '男' and P_Salary >= 10000

--计算出在中国的男性员工的最大年龄，最小年龄和平均年龄(方法一)
select '男' Gender,COUNT(*) Count_Num,
 MAX(year(getdate())-year(P_Birthday)) Max_P_Birthady,
 MIN(year(getdate())-year(P_Birthday)) Min_P_Birthady,
 SUM(year(getdate())-year(P_Birthday)) Sum_P_Birthady,
 AVG(year(getdate())-year(P_Birthday)) Avg_P_Birthady

--计算出在中国的男性员工的最大年龄，最小年龄和平均年龄(方法二)
select '男' Gender,COUNT(*) Count_Num,
 MAX(DATEDIFF(year, P_Birthday, getDate())) Max_P_Birthady,
 MIN(DATEDIFF(year, P_Birthday, getDate())) Min_P_Birthady,
 SUM(DATEDIFF(year, P_Birthday, getDate())) Sum_P_Birthady,

--求出年龄比平均年龄高的人员信息(方法一)
select * from People where 
 year(getdate())-year(P_Birthday) >
 (select AVG(year(getdate())-year(P_Birthday)) 
from People)

--求出年龄比平均年龄高的人员信息(方法二)
select * from People where 
 DATEDIFF(year, P_Birthday, getDate()) >
 (select AVG(DATEDIFF(year, P_Birthday, getDate())) 
from People)

其中，ROUND函数用法：

round(num,len,[type])
num表示需要处理的数字，len表示需要保留的长度，type处理类型(0是默认值代表四舍五入，非0代表直接截取)
select ROUND(123.45454,3) –123.45500
select ROUND(123.45454,3,1) –123.45400

分组查询

--选取数据库
use DBtest

--根据员工所在地区分组统计员工人数 ,员工工资总和 ,平均工资，最高工资和最低工资
--方法一：使用union(此方案需要知道所有的地区，分别查询出所有地区的数据，然后使用union拼接起来。)
select '武汉' 'city or country',COUNT(*) Count_Num,MAX(P_Salary) Max_P_Salary,MIN(P_Salary) Min_P_Salary,SUM(P_Salary) Sum_P_Salary,AVG(P_Salary) Avg_P_Salary 
from People 
where P_Address = '武汉' 
union
select '中国' 'city or country',COUNT(*) Count_Num,MAX(P_Salary) Max_P_Salary,MIN(P_Salary) Min_P_Salary,SUM(P_Salary) Sum_P_Salary,AVG(P_Salary) Avg_P_Salary 
from People 
where P_Address = '中国' 
 --...其它地区数据
 --...其他地区数据
 --方法二：使用Group by
select P_Address 'city or country',COUNT(*) Count_Num,MAX(P_Salary) Max_P_Salary,MIN(P_Salary) Min_P_Salary,SUM(P_Salary) Sum_P_Salary,AVG(P_Salary) Avg_P_Salary 
from People 
group by P_Address

--根据员工所在地区分组统计员工人数，员工工资总和，平均工资，最高工资和最低工资，1996 年及以后出身的员工不参与统计。
select P_Address 'city or country',COUNT(*) Count_Num,MAX(P_Salary) Max_P_Salary,MIN(P_Salary) Min_P_Salary,SUM(P_Salary) Sum_P_Salary,AVG(P_Salary) Avg_P_Salary 
from People
where P_Birthday < '1996-1-1'
group by P_Address

--根据员工所在地区分组统计员工人数，员工工资总和，平均工资，最高工资和最低工资，要求筛选出员工人数至少在1人及以上的记录，并且1996年及以后出身的员工不参与统计。
select P_Address 'city or country',COUNT(*) Count_Num,MAX(P_Salary) Max_P_Salary,MIN(P_Salary) Min_P_Salary,SUM(P_Salary) Sum_P_Salary,AVG(P_Salary) Avg_P_Salary 
from People 
where P_Birthday < '1996-1-1'
group by P_Address
having COUNT(*) >= 1

多表查询

--选取数据库
use DBtest

--笛卡尔乘积
select * from People
select * from Department
select * from People,Department
--此查询结果会将People表的所有数据和Department表的所有数据进行依次排列组合形成新的记录。

--简单多表查询:此种查询不符合主外建关系的数据不会被显示
--查询员工信息，同时显示部门名称
select * from People,Department where People.D_Id = Department.D_Id 
--查询员工信息，同时显示职级名称
select * from People,[Rank] where People.R_Id = [Rank].R_Id 
--查询员工信息，同时显示部门名称，职位名称

--右外连接（right join）：右外连接和左外连接类似，A left join B == B right join A
--全外连接（full join）：两张表的所有数据无论是否符合主外键关系必须全部显示，不符合主外键关系的地方null取代。

表设计:

1.账户信息表：存储个人信息。
2.银行卡表：存储银行卡信息。
3.交易信息表（存储存钱和取钱的记录）
4.转账信息表（存储转账信息记录）
5.状态信息变化表（存储银行卡状态变化的记录，状态有1:正常,2:挂失,3:冻结,4:注销）

--账户信息表：存储个人信息
create table AccountInfo
(
    A_Id int primary key identity(1,1), --账户编号
    A_Code varchar(20) not null, --身份证号码
    A_Phone varchar(20) not null, --电话号码
    A_RealName varchar(20) not null, --真实姓名
    A_OpenTime smalldatetime not null, --开户时间
)
--银行卡表：存储银行卡信息
create table BankCard
(
    B_CardNo varchar(30) primary key, --银行卡卡号
    B_A_Id int references AccountInfo(A_Id) , --账户编号（与账户信息表形成主外键关系）
    B_Pwd varchar(30) not null, --银行卡密码
    B_Money money not null, --银行卡余额
    B_State int not null,--1:正常,2:挂失,3:冻结,4:注销
    B_Time smalldatetime default(getdate()) --开卡时间
)
--交易信息表（存储存钱和取钱的记录）
create table CardExchange
(
    C_ExchangeId int primary key identity(1,1), --交易自动编号
    C_No varchar(30) references BankCard(B_CardNo), --银行卡号(与银行卡表形成主外键关系)
    C_MoneyInBank money not null, --存钱金额
    C_MoneyOutBank money not null, --取钱金额
    C_ExchangeTime smalldatetime not null, --交易时间
)
--转账信息表（存储转账信息记录）
create table CardTransfer
(
    C_TransferId int primary key identity(1,1),--转账自动编号
    C_NoOut varchar(30) references BankCard(B_CardNo), --转出银行卡号(与银行卡表形成主外键关系)
    C_NoIn varchar(30) references BankCard(B_CardNo), --转入银行卡号(与银行卡表形成主外键关系)
    C_TransferMoney money not null,--交易金额
    C_TransferTime smalldatetime not null, --交易时间
)
--状态信息变化表（存储银行卡状态变化的记录，状态有1:正常,2:挂失,3:冻结,4:注销）
create table CardStateChange
(
    C_StateId int primary key identity(1,1),--状态信息自动编号
    C_No varchar(30) references BankCard(B_CardNo), --银行卡号(与银行卡表形成主外键关系)
    C_OldState int not null, --银行卡原始状态
    C_NewState int not null, --银行卡新状态
    C_StateWhy varchar(200) not null, --状态变化原因
    C_StateTime smalldatetime not null, --记录产生时间
)

变量

T-SQL中变量分为局部变量和全局变量

局部变量：（1）以@作为前缀（2）先声明，在赋值

declare @str varchar(20)
set @str = 'BingoLin' --或者select @str = 'BingoLin'
print @str

备注：set赋值和select赋值区别：

set：赋给变量指定的值

select：一般用于从表中查询出的数据，查询记录如果有多条，将最后一条记录的值赋给变量，

例如：

select @变量名 = 字段名 from 表名

在赋值过程中，如果是表中查询的数据，如果记录只有一条，使用set和select都可以，但是习惯上使用select。

全局变量：（1）以@@作为前缀（2）由系统进行定义和维护,只读

@@ERROR：返回执行的上一个语句的错误号

@@IDENTITY：返回最后插入的标识值

@@MAX_CONNECTIONS：返回允许同时进行的最大用户连接数

@@ROWCOUNT：返回受上一语句影响的行数

@@SERVERNAME：返回运行 SQL Server 的本地服务器的名称

@@SERVICENAME：返回 SQL Server 正在其下运行的注册表项的名称

@@TRANCOUNT：返回当前连接的活动事务数

@@LOCK_TIMEOUT：返回当前会话的当前锁定超时设置（毫秒）

二、循环结构

（1）循环打印1-10。

declare @i int  = 1
while @i <= 10
begin
    print @i
    set @i = @i + 1
end

（2）循环打印九九乘法表

declare @i int = 1
declare @str varchar(1000)
while @i<=9
begin
    declare @j int = 1
    set @str = ''
    while @j <= @i
    begin
        --方案一
        --set @str = @str + cast(@i as varchar(2)) + '*' + cast(@j as varchar(2)) + 
        --'=' + cast(@i*@j as varchar(2)) + CHAR(9)
        --方案二
        set @str = @str + Convert(varchar(2),@i) + '*' + Convert(varchar(2),@j) + 
        '=' + Convert(varchar(2),@i*@j) + CHAR(9)       
        set @j = @j + 1
    end
    print @str
    set @i = @i + 1
end

备注：

（1）特殊字符：制表符 CHAR(9)；换行符 CHAR(10)；回车 CHAR(13)；

（2）循环中若出现break和CONTINUE,作用与Java，C#等语言一致。
