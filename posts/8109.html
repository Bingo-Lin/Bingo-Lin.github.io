创建表

if exists(select*from sys.objects where name = 'Department' and type = 'U')
 drop table Department --超级危险，慎用

use DBtest --切换数据库

--创建表基本语法
create table Department(
 D_Id int primary key identity(1,1), --identity(1,1):自动增长，初始值1，每次增长1。
 D_Name nvarchar(50) not null,
 D_Remark text --部门描述
)    
--char:定长
--varchar:变长
--text:长文本(效率较低)
--char,varchar,text前面加n:存储unicode字符，对中文比较友好
--比如：varchar(100):存储100个字母或者50个汉字
--nvarchar(100):存储100个字母或者100个汉字

create table [Rank](
 R_Id int primary key identity(1,1), --identity(1,1):自动增长，初始值1，每次增长1。
 R_Name nvarchar(50) not null,
 R_Remark text     
)  

create table People (
 P_Id int primary key identity(1,1), --identity(1,1):自动增长，初始值1，每次增长1。
 D_Id int references Department(D_Id) not null , --引用外键
 R_Id int references [Rank](R_Id) not null ,   
 P_Name nvarchar(50) not null,
 P_Gender nvarchar(1) default('男') check(P_Gender = '男'or P_Gender = '女') not null ,
 P_Birthday smalldatetime not null,
 P_Salary decimal(12,2) check(P_Salary >= 1000 and  P_Salary <= 1000000) not null ,
 P_Phone varchar(20) unique not null,
 P_Address varchar(300),
 P_AddTime smalldatetime default(getdate())
)

维护约束(删除，添加)

--删除约束
--语法：alter table tableName drop constraint constrantName
--例子：删除月薪的约束
alter table People drop constraint CK__People__P_Salary__38996AB5

--添加约束(check约束)
alter table tableName add constraint constraintName check (表达式) 
--例子：添加工资字段约束，工资必须在1000-1000000之间
alter table People add constraint CK__People__P_Salary check (P_Salary >= 1000 and P_Salary <= 1000000)

--添加约束(主键约束) 语法：
alter table tableName add constraint constraintName  primary key(列名)

--添加约束(唯一约束) 语法：
alter table tableName add constraint constraintName  unique(列名)

--添加约束(默认值) 语法：
alter table tableName add constraint constraintName  default 默认值 for 列名 

--添加约束(外键) 语法：
alter table tableName add constraint constraintName  foreign key(列名) references 关联表名(列名)

drop,truncate,delete简单区别：

--关于删除(drop,truncate,delete)
delete from People --删除表数据，表对象和表结构依然存在
truncate table People --删除表数据(清空数据)，表对象和表结构依然存在
drop table People --删除表对象(删除最彻底)
--truncate ,delete 区别：
--truncate 清空所有数据，不能有条件。同时已经自动编号1，2，3，4 ...仍存在。即，原先表中有1，2，3，4.然后truncate 后，再添加数据，自动编号仍是1234
--delete 可以删除所有数据，也可以带条件删除符合条件的数据。同时，已经自动编号的编号将永远消失。即，原先表中有1，2，3，4.然后delete 后，再添加数据，自动编号将变成5，6，7，8

模糊查询

%：代表匹配0个字符、1个字符或多个字符
_：代表匹配有且只有1个字符
[]：代表匹配范围内
[^]：代表匹配不在范围内

例子如下：

--选取数据库
use DBtest

--查询姓刘的员工信息
select * from People  where P_Name like '刘%'

--查询名字中含有'备'的员工信息
select * from People  where P_Name like '%备%'

--查询名字中含有'备'或者'刘'的员工信息
select * from People  where P_Name like '%备%' or P_Name like '%刘%'

--查询姓刘的员工，名字是2个字(方法一)
select * from People  where P_Name like '刘_'

--查询姓刘的员工，名字是2个字(方法二)
select * from People where substring(P_Name,1,1) = '刘' and len(P_Name) = 2

--查询出名字最后一个字是备，名字一共两个字的员工信息(方法一)
select * from People  where P_Name like '_备'

--查询出名字最后一个字是备，名字一共两个字的员工信息(方法二)
select * from People where substring(P_Name,2,1) = '备' and LEN(P_Name) = 2

--查询出电话号码开头138的员工信息
select * from People where P_Phone like '138%'

--查询出电话号码开头138的员工信息,第4位可能是5，可能8 ，最后一个号码是8
select * from People where P_Phone like '138[5,8]%8'

--查询出电话号码开头138的员工信息,第4位是2-5之间的数字 ，最后一个号码不是2和3(方法一)
select * from People where P_Phone like '138[2,3,4,5]%[^2,3]'

--查询出电话号码开头138的员工信息,第4位是2-5之间的数字 ，最后一个号码不是2和3(方法二)
select * from People where P_Phone like '138[2-5]%[^2-3]'

索引

索引：提高检索查询效率。

SQL SERVER索引类型：按存储结构区分：“聚集索引（又称聚类索引，簇集索引）”，“非聚集索引（非聚类索引，非簇集索引）”；

聚集索引：根据数据行的键值在表或视图中的排序存储这些数据行，每个表只有一个聚集索引。聚集索引是一种对磁盘上实际数据重新组织以按指定的一列或多列值排序(类似字典中的拼音索引)(物理存储顺序)。

非聚集索引：具有独立于数据行的结构，包含非聚集索引键值，且每个键值项都有指向包含该键值的数据行的指针。（类似字典中的偏旁部首索引）（逻辑存储顺序）。

SQL SERVER索引其他分类：

按数据唯一性区分：“唯一索引”，“非唯一索引”；按键列个数区分：“单列索引”，“多列索引”。

创建索引的方式：

  1. 通过显式的CREATE INDEX命令
  2. 在创建约束时作为隐含的对象
    1. 主键约束(聚集索引)
    2. 唯一约束(唯一索引)

创建索引语法：

CREATE [UNIQUE] [CLUSTERED | NONCLUSTERED]
INDEX <index name> ON <table or view name>(<column name> [ASC|DESC][,...n])

    from BankCard inner join AccountInfo on BankCard.B_A_Id = AccountInfo.A_Id

模拟银行卡取钱操作，传入银行卡号，取钱金额，实现取钱操作，取钱成功，返回1，取钱失败返回-1

create proc proc_QuQian
@CardNo varchar(30),
@MoneyOutBank money
as
    update BankCard set B_Money = B_Money - @MoneyOutBank where B_CardNo = @CardNo
    if @@ERROR <> 0
        return -1
    insert into CardExchange(C_No,C_MoneyInBank,C_MoneyOutBank,C_ExchangeTime)
    values(@CardNo,0,@MoneyOutBank,GETDATE())
    return 1
go

密码升级，传入用户名和密码，如果用户名密码正确，并且密码长度<8，自动升级成8位密码

--有输入输出参数（密码作为输入参数也作为输出参数）
--密码升级，传入用户名和密码，如果用户名密码正确，并且密码长度<8，自动升级成8位密码

select FLOOR(RAND()*10) --0-9之间随机数
go
create proc proc_PwdUpgrade
@cardno nvarchar(20),
@pwd nvarchar(20) output
as
    if not exists(select * from BankCard where B_A_Id=@cardno and B_Pwd=@pwd)
        set @pwd = ''
    else
    begin
        if len(@pwd) < 8
        begin
            declare @len int = 8- len(@pwd)
            declare @i int = 1
            while @i <= @len
            begin

                set @pwd = @pwd + cast(FLOOR(RAND()*10) as varchar(1))
                set @i = @i+1
            end
            update BankCard set B_Pwd = @pwd where B_CardNo=@cardno
        end
    end
go
--执行
declare @pwd nvarchar(20) = '123456'
exec proc_PwdUpgrade '6225547854125656',@pwd output
select @pwd
